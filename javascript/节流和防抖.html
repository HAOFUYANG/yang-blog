# debounce

function debounce(func, delay=500) {
  let timer = null
  return function(){
    if(timer){
      clearTimeout(timer)
    }
    timer = setTimeout(()=>{
      func.apply(this,arguments)
    },delay)
  }
}
function handle() {
    console.log('防抖：', Math.random());
}  
window.addEventListener('scroll', debounce(handle,500))
